28                  AerialParamKeys.INPUT_TYPE.toString(),

45      private void runPredefinedMethods(Class<?> annotation) throws Exception {
46          if (!annotation.isAnnotation()) {
47              return;
48          }
49          Method[] methodList = this.clazz.getMethods();
50          for (Method method : methodList) {
51              Annotation[] annotations = method.getAnnotations();
52              for (Annotation item : annotations) {
53                  if (item.annotationType().equals(annotation)) {
54                      method.invoke(null);
55                      break;
56                  }
57              }
58          }
59      }
