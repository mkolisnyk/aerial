31      private Map<String, List<String>> filterTestData(Map<String, List<String>> testData, boolean positive) {
32          Map<String, List<String>> result = new HashMap<String, List<String>>();
33          int count = testData.get(testData.keySet().iterator().next()).size();
34          for (int i = 0; i < count; i++) {
35              if (testData.get("ValidInput").get(i).trim().equals("false") == positive) {
36                  continue;
37              }
38              for (Entry<String, List<String>> entry:testData.entrySet()) {
39                  if (!result.containsKey(entry.getKey())) {
40                      result.put(entry.getKey(), new ArrayList<String>());
41                  }
42                  result.get(entry.getKey()).add(testData.get(entry.getKey()).get(i));
43              }
44          }
45          return result;
46      }

48      private String generateTestData(Map<String, List<String>> testData) throws IOException {
49          String dataHeader = AerialOutputTemplateMap.get(AerialOutputFormat.getCurrent().toString(), "data.header");
50          String dataHeaderDelimiter = AerialOutputTemplateMap.get(
51                  AerialOutputFormat.getCurrent().toString(), "data.header.delimiter");
52          String dataRow = AerialOutputTemplateMap.get(AerialOutputFormat.getCurrent().toString(), "data.row");
53          String dataRowDelimiter = AerialOutputTemplateMap.get(
54                  AerialOutputFormat.getCurrent().toString(), "data.row.delimiter");
55          String content = dataHeader.replaceAll(
56                  "\\{TITLES\\}",
57                  StringUtils.join(testData.keySet().iterator(), dataHeaderDelimiter)
58          );
59          int count = testData.get(testData.keySet().iterator().next()).size();
60          for (int i = 0; i < count; i++) {
61              String dataRowText = "";
62              String[] row = new String[testData.entrySet().size()];
63              int j = 0;
64              for (Entry<String, List<String>> entry : testData.entrySet()) {
65                  row[j] = entry.getValue().get(i);
66                  j++;
67              }
68              dataRowText = StringUtils.join(row, dataRowDelimiter);
69              dataRowText = dataRow.replaceAll("\\{DATA\\}", dataRowText);
70              content = content.concat(dataRowText);
71          }
72          return content;
73      }

79      @Override
80      public String generate() throws Exception {
81          String template = AerialOutputTemplateMap.get(AerialOutputFormat.getCurrent().toString(), "case");
82  
83          String content = "";
84          content += this.generatePreRequisites(this.getSection().getSections().get(Tokens.getPrerequisitesToken()));
85          content += this.getSection().getSections().get(Tokens.getActionToken()).get(0).generate();
86          if (isPositive()) {
87              content += this.getSection().getSections().get(Tokens.getValidOutputToken()).get(0).generate();
88          } else {
89              content += this.getSection().getSections().get(Tokens.getErrorOutputToken()).get(0).generate();
90          }
91          Map<String, List<String>> testData = filterTestData(this.getTestData(), isPositive());
92          if (isPositive()) {
93              NWiseDataTestingAlgorithm algorithm = new NWiseDataTestingAlgorithm(testData, DEFAULT_RECORD_SIZE);
94              testData = algorithm.generateTestData();
95          }
96          String result = template.replaceAll("\\{NAME\\}", this.getSection().getName() + this.getScenarioName())
97                              .replaceAll("\\{BODY\\}", content)
98                              .replaceAll("\\{DATA\\}", this.generateTestData(testData))
99                              .replaceAll("\\{TAGS\\}", this.getTagsString());
100         return result;
101     }
