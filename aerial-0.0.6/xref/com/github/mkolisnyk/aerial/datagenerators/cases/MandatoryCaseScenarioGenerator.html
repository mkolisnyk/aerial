29      @Override
30      public String generate() throws Exception {
31          String template = AerialOutputTemplateMap.get(AerialOutputFormat.getCurrent().toString(), "case");
32  
33          String[] uniqueRecords = getFieldsWithMandatoryAttributes(this.getRecords());
34          String content = "";
35          content += this.generatePreRequisites(this.getSection().getSections().get(Tokens.getPrerequisitesToken()));
36          content += this.getSection().getSections().get(Tokens.getActionToken()).get(0).generate();
37          content += this.getSection().getSections().get(Tokens.getErrorOutputToken()).get(0).generate();
38          String result = template.replaceAll("\\{NAME\\}", this.getSection().getName() + this.getScenarioName())
39                  .replaceAll("\\{BODY\\}", content)
40                  .replaceAll("\\{DATA\\}", this.generateMandatoryScenarioData(this.getTestData(), uniqueRecords))
41                  .replaceAll("\\{TAGS\\}", this.getTagsString());
42          return result;

46      private String generateMandatoryScenarioData(
47              Map<String, List<String>> testData, String[] mandatoryFields) throws Exception {
48          Map<String, List<String>> filteredData = filterBy(testData, "ValidInput", "true");
49          String dataHeader = AerialOutputTemplateMap.get(AerialOutputFormat.getCurrent().toString(), "data.header");
50          String dataHeaderDelimiter = AerialOutputTemplateMap.get(
51                  AerialOutputFormat.getCurrent().toString(), "data.header.delimiter");
52          String dataRow = AerialOutputTemplateMap.get(AerialOutputFormat.getCurrent().toString(), "data.row");
53          String dataRowDelimiter = AerialOutputTemplateMap.get(
54                  AerialOutputFormat.getCurrent().toString(), "data.row.delimiter");
55  
56          Map<String, String> filteredRow = new LinkedHashMap<String, String>();
57          for (Entry<String, List<String>> entry : filteredData.entrySet()) {
58              filteredRow.put(entry.getKey(), entry.getValue().get(0));
59          }
60          String content = "";
61          content = dataHeader.replaceAll(
62                  "\\{TITLES\\}",
63                  StringUtils.join(filteredRow.keySet().iterator(), dataHeaderDelimiter)
64          );
65          for (String field : mandatoryFields) {
66              String[] row = new String[filteredRow.entrySet().size()];
67              int i = 0;
68              for (Entry<String, String> entry : filteredRow.entrySet()) {
69                  if (entry.getKey().equals(field)) {
70                      row[i++] = "";
71                  } else {
72                      row[i++] = entry.getValue();
73                  }
74              }
75              String dataRowText = StringUtils.join(row, dataRowDelimiter);
76              dataRowText = dataRow.replaceAll("\\{DATA\\}", dataRowText);
77              content = content.concat(dataRowText);
78          }
79          return content;
80      }

82      private String[] getFieldsWithMandatoryAttributes(
83              List<InputRecord> input) {
84          Set<String> fields = new HashSet<String>();
85          String[] result;
86          for (InputRecord record : input) {
87              if (record.isMandatory()) {
88                  fields.add(record.getName());
89              }
90          }
91          result = new String[fields.size()];
92          result = fields.toArray(result);
93          return result;
94      }
