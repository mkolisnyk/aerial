48      private DocumentSection<?> createSection(String token) throws Exception {
49          Map<String, Class<?>> creationMap = getCreationMap();
50          if (creationMap.containsKey(token)) {
51              Class<?> clazz = creationMap.get(token);
52              return (DocumentSection<?>) clazz
53                          .getConstructor(DocumentSection.class, String.class)
54                          .newInstance(this, this.getTag());
55          }
56          return null;
57      }

59      private void setCurrentSection(
60              String currentSectionName,
61              DocumentSection<?> currentSection,
62              String content) throws Exception {
63          if (currentSection == null) {
64              this.description = content;
65              content = "";
66          } else {
67              ArrayList<DocumentSection<?>> sectionItem = new ArrayList<DocumentSection<?>>();
68              if (this.sections.containsKey(currentSectionName)) {
69                  sectionItem = this.sections.get(currentSectionName);
70              }
71              sectionItem.add(currentSection.parse(content));
72              this.sections.put(
73                      currentSectionName,
74                      sectionItem);
75          }
76      }
77      public ContainerSection parse(String input) throws Exception {
78          String currentSectionName = "";
79          DocumentSection<?> currentSection = null;
80          String content = "";
81          String separator = System.lineSeparator();
82  
83          for (String line:input.split(separator)) {
84              boolean isToken = false;
85              String tokenFound = "";
86              for (String token : this.getSectionTokens()) {
87                  if (line.trim().startsWith(token)) {
88                      isToken = true;
89                      tokenFound = token;
90                      break;
91                  }
92              }
93              if (isToken) {
94                  setCurrentSection(currentSectionName, currentSection, content);
95                  currentSectionName = tokenFound;
96                  currentSection = createSection(tokenFound);
97                  String trimLine = line.trim();
98                  if (trimLine.equals(tokenFound)) {
99                      currentSection.setName("");
100                 } else {
101                     currentSection.setName(trimLine.substring(tokenFound.length()));
102                 }
103                 content = "";
104             } else {
105                 if (content.equals("")) {
106                     content = content.concat(line);
107                 } else {
108                     content = content.concat(separator + line);
109                 }
110             }
111         }
112         setCurrentSection(currentSectionName, currentSection, content);
113         return this;
114     }

116     public void validate() throws Exception {
117         for (String token:this.getMandatoryTokens()) {
118             Assert.assertTrue(this.sections.containsKey(token));
119             for (DocumentSection<?> item : this.sections.get(token)) {
120                 item.validate();
121             }
122         }
123         for (String token:this.getSectionTokens()) {
124             if (!ArrayUtils.contains(this.getMandatoryTokens(), token)
125                     && this.sections.containsKey(token)) {
126                 for (DocumentSection<?> item : this.sections.get(token)) {
127                     item.validate();
128                 }
129             }
130         }
