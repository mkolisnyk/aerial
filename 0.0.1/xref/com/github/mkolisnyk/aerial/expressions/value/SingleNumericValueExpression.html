40      private InputRecord nonCurrentValue(InputRecord input) {
41          InputRecord testRecord;
42          Integer intValue = new Integer(input.getValue());
43          String updatedValue = "";
44          if (intValue == 0) {
45              updatedValue = "-1";
46          } else {
47              updatedValue = "0";
48          }
49          testRecord = new InputRecord(
50                  input.getName(),
51                  input.getType(),
52                  updatedValue,
53                  input.getCondition(),
54                  false);
55          return testRecord;
56      }
