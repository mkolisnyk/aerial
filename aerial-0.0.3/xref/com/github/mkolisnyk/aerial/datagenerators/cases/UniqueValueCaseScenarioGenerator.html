31      private String[] getFieldsWithUniqueAttributes(
32              List<InputRecord> input) {
33          Set<String> fields = new HashSet<String>();
34          String[] result;
35          for (InputRecord record : input) {
36              if (record.isUnique()) {
37                  fields.add(record.getName());
38              }
39          }
40          result = new String[fields.size()];
41          result = fields.toArray(result);
42          return result;
43      }

45      Map<String, List<String>> filterBy(
46              Map<String, List<String>> testData, String field,
47              String value) {
48          Map<String, List<String>> result = new HashMap<String, List<String>>();
49          for (String key : testData.keySet()) {
50              result.put(key, new ArrayList<String>());
51          }
52          int count = testData.get(testData.keySet().iterator().next())
53                  .size();
54          for (int i = 0; i < count; i++) {
55              if (testData.get(field).get(i).trim().equals(value)) {
56                  for (String key : testData.keySet()) {
57                      result.get(key).add(testData.get(key).get(i));
58                  }
59              }
60          }
61          return result;
62      }

64      private String getValueDifferentFrom(List<String> input,
65              String value) {
66          for (String item : input) {
67              if (!item.equals(value)) {
68                  return item;
69              }
70          }
71          return value;
72      }

74      private String generateUniqueScenarioData(
75              Map<String, List<String>> testData, String[] uniqueFields) throws IOException {
76          Map<String, List<String>> filteredData = filterBy(testData, "ValidInput", "true");
77          Map<String, String> uniqueRow = new LinkedHashMap<String, String>();
78          String dataHeader = AerialTemplateMap.get(AerialOutputFormat.getCurrent().toString(), "data.header");
79          String dataHeaderDelimiter = AerialTemplateMap.get(
80                  AerialOutputFormat.getCurrent().toString(), "data.header.delimiter");
81          String dataRow = AerialTemplateMap.get(AerialOutputFormat.getCurrent().toString(), "data.row");
82          String dataRowDelimiter = AerialTemplateMap.get(
83                  AerialOutputFormat.getCurrent().toString(), "data.row.delimiter");
84          String modifiedPrefix = AerialTemplateMap.get(
85                  AerialOutputFormat.getCurrent().toString(), "data.field.modified_prefix");
86  
87          for (Entry<String, List<String>> entry : filteredData.entrySet()) {
88              String value = entry.getValue().get(0);
89              uniqueRow.put(entry.getKey(), value);
90              if (ArrayUtils.contains(uniqueFields, entry.getKey())) {
91                  uniqueRow.put(modifiedPrefix + entry.getKey(), value);
92              }
93          }
94          String content = dataHeader.replaceAll(
95                  "\\{TITLES\\}",
96                  StringUtils.join(uniqueRow.keySet().iterator(), dataHeaderDelimiter)
97          );
98  
99          for (String field : uniqueFields) {
100             String dataRowText = "";
101             String[] row = new String[uniqueRow.entrySet().size()];
102             int j = 0;
103 
104             for (Entry<String, String> entry : uniqueRow.entrySet()) {
105                 String key = entry.getKey();
106                 String value = entry.getValue();
107                 if (key.startsWith(modifiedPrefix)) {
108                     String modField = key.replaceFirst(modifiedPrefix, "");
109                     if (!modField.equals(field)) {
110                         value = getValueDifferentFrom(
111                                 filteredData.get(modField), value);
112                     }
113                 } else {
114                     if (!key.equalsIgnoreCase(field)
115                             && !ArrayUtils
116                                     .contains(uniqueFields, key)) {
117                         value = getValueDifferentFrom(
118                                 filteredData.get(key), value);
119                     }
120                 }
121                 row[j] = value.trim();
122                 j++;
123             }
124             dataRowText = StringUtils.join(row, dataRowDelimiter);
125             dataRowText = dataRow.replaceAll("\\{DATA\\}", dataRowText);
126             content = content.concat(dataRowText);
127         }
128         return content;
129     }

131     @Override
132     public String generate() throws Exception {
133         String template = AerialTemplateMap.get(AerialOutputFormat.getCurrent().toString(), "case");
134         String modifiedPrefix = AerialTemplateMap.get(
135                 AerialOutputFormat.getCurrent().toString(), "data.field.modified_prefix");
136         String fieldPattern = AerialTemplateMap.get(
137                 AerialOutputFormat.getCurrent().toString(), "data.field");
138 
139         String[] uniqueRecords = getFieldsWithUniqueAttributes(this.getRecords());
140         String content = "";
141         content += this.generatePreRequisites(this.getSection().getSections().get(Tokens.PREREQUISITES_TOKEN));
142         content += this.getSection().getSections().get(Tokens.ACTION_TOKEN).get(0).generate();
143         content += this.getSection().getSections().get(Tokens.VALID_OUTPUT_TOKEN).get(0).generate();
144         String secondPass = this.getSection().getSections().get(Tokens.ACTION_TOKEN).get(0).generate()
145                 + this.getSection().getSections().get(Tokens.ERROR_OUTPUT_TOKEN).get(0).generate();
146         for (String field : uniqueRecords) {
147             secondPass = secondPass.replaceAll(fieldPattern.replaceAll("\\{NAME\\}", field),
148                     fieldPattern.replaceAll("\\{NAME\\}", modifiedPrefix + field));
149         }
150         content += secondPass;
151         String result = template.replaceAll("\\{NAME\\}", this.getSection().getName() + this.getScenarioName())
152                 .replaceAll("\\{BODY\\}", content)
153                 .replaceAll("\\{DATA\\}", this.generateUniqueScenarioData(this.getTestData(), uniqueRecords));
154         return result;
155     }
